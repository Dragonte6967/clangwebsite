				sum += a[i][j];

		}
	}

	printf("%d", sum);
}

#include <stdio.h>

int main()
{
	int m, n, sum = 0,count=0;
	scanf("%d%d", &m, &n);
	int a[m][n];

	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			scanf("%d", &a[i][j]);
		}
	}

	for (int i = 0; i < m; i++)
	{
		for (int j = 0; j < n; j++)
		{
			if (a[i][j] % 2 == 0)
			{
				sum += a[i][j];
				count++;
			}
		}
	}
	if (count !=0)
	{
		printf("Average = %.2f", 1.0 * sum / count);
	}else{
		printf("Average = 0");
	}
	
	return 0;
}

#include <stdio.h>
int main()
{
	int m, n, i, t, count = 0;
	scanf("%d %d", &m, &n);
	int a[m][n];
	for (i = 0; i < m * n; i++)
		scanf("%d", &a[0][i]);
	for (i = 1; i < m - 1; i++)
		for (t = 1; t < n - 1; t++)
		{
			if (a[i][t] > a[i - 1][t] && a[i][t] > a[i][t - 1] && a[i][t] > a[i + 1][t] && a[i][t] > a[i][t + 1])
			{
				printf("%d %d %d\n", a[i][t], i + 1, t + 1);
				count++;
			}
		}
	if (count == 0)
		printf("None %d %d", m, n);
	return 0;
}

int main()
{
	int T, n, i = 0;
	scanf("%d", &T);
	while (1)
	{
		scanf("%d", &n);
		i++;
		int matrix[n][n];
		for (int j = 0; j < n; j++)
		{
			for (int k = 0; k < n; k++)
			{
				scanf("%d", &matrix[j][k]);
			}
		}
		int flag = 0;
		for (int j = 0; j < n; j++)
		{
			for (int k = 0; k < j; k++)
			{
				if (matrix[j][k] != 0)
				{
					flag = 1;
				}
			}
		}
		if (flag)
		{
			printf("NO\n");
		}
		else
		{
			printf("YES\n");
		}
		if (i >= T)
		{
			break;
		}
	}
	return 0;
}
