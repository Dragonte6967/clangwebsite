#include <stdio.h>
// #include <math.h>

int main()
{
    int a, b, c;
    float area, peri, s;
    scanf("%d %d %d", &a, &b, &c);
    if (a + b > c && a + c > b && b + c > a)
    {
        s = 0.5 * (a + b + c);
        area = sqrt(s * (s - a) * (s - b) * (s - c));
        peri = 1.00 * (a + b + c);
        printf("area = %.2f; perimeter = %.2f", area, peri);
    }
    else
    {
        printf("These sides do not correspond to a valid triangle");
    }
    return 0;
}

## 7.3 

### `switch`写法
```clike
#include <stdio.h>

int main()
{
    int i;
    scanf("%d", &i);
    switch (i % 4)
    {
    case 0:
        printf("answer = A");
        break;
    case 1:
        printf("answer = B");
        break;
    case 2:
        printf("answer = C");
        break;
    case 3:
        printf("answer = D");
        break;
        return 0;
    }
}
```
### `if-else`写法

```clike
#include <stdio.h>

int main()
{
    int i;
    scanf("%d", &i);
    if(i % 4 ==0)
        printf("answer = A");
    if(i % 4 == 1)
        printf("answer = B");
    if(i % 4 == 2)
        printf("answer = C");
    if(i % 4 == 3)
        printf("answer = D");
    return 0;
}
```

int main()
{
    int a;
    int b;
    scanf("%d %d", &a, &b);
    if (b > 0)
        printf("%d/%d=%.2lf", a, b, 1.0 * a / b);
    else if (b == 0)
        printf("%d/%d=Error", a, b);
    else
        printf("%d/(%d)=%.2lf", a, b, 1.0 * a / b);
